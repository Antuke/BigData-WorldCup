# avviare HADOOP e YARN se non sono già avviati
$HADOOP_HOME/sbin/start-dfs.sh
$HADOOP_HOME/sbin/start-yarn.sh

# Inserisco il dataset nel dfs
hdfs dfs -put worldcup1.csv /dataset

# Lancio Hadoop jar
# sostituire $0 con le iniziali di una nazione e.g ITA
hadoop jar FinalProject.jar /dataset /refereeIndex $0

# Prendo dal dfs il risultato di hadoop
hdfs dfs -get /refereeIndex 

# Pulizia del dfs (cancello il dataset e il risultato di Hadoop)
hdfs dfs -rm -r /refereeIndex
hdfs dfs -rm /dataset
